00084: class TaskChangeObservable {
00085:   private List<TaskChangeObserver> observers = new ArrayList<TaskChangeObserver>();
00086: 
00087:   public void addTaskChangeObserver(TaskChangeObserver observer) {
00088:     if (!observers.contains(observer)) {
00089:       observers.add(observer);
00090:     }
00091:   }
00092: 
00093:   public void removeTaskChangeObserver(TaskChangeObserver observer) {
00094:     observers.remove(observer);
00095:   }
00096: 
00097:   public void selectTask(Task task) {
00098:     for (TaskChangeObserver o : observers) {
00099:       o.taskSelected(task);
00100:     }
00101:   }
00102: 
00103:   public void addTask(Task task) {
00104:     for (TaskChangeObserver o : observers) {
00105:       o.taskSelected(task);
00106:     }
00107:   }
00108: 
00109:   public void updateTask(Task task) {
00110:     for (TaskChangeObserver o : observers) {
00111:       o.taskChanged(task);
00112:     }
00113:   }
00114: }

00168:   public void setTaskChangeObservable(TaskChangeObservable newNotifier) {
00169:     notifier = newNotifier;

00171: 
00172:   public void actionPerformed(ActionEvent event) {
00173:     Object source = event.getSource();
00174:     if (source == add) {
00175:       double timeRequired = 0.0;
00176:       try {
00177:         timeRequired = Double.parseDouble(taskTime.getText());
00178:       } catch (NumberFormatException exc) {
00179:       }
00180:       notifier.addTask(new Task(taskName.getText(), taskNotes.getText(),
00181:           timeRequired));
00182:     } else if (source == update) {
00183:       editTask.setName(taskName.getText());
00184:       editTask.setNotes(taskNotes.getText());
00185:       try {
00186:         editTask
00187:             .setTimeRequired(Double.parseDouble(taskTime.getText()));
00188:       } catch (NumberFormatException exc) {
00189:       }
00190:       notifier.updateTask(editTask);
00191:     } else if (source == exit) {
00192:       System.exit(0);
00193:     }
00194: 
00195:   }

00198:   }

00201:   }

00206:     taskNotes.setText(task.getNotes());
00207:     taskTime.setText("" + task.getTimeRequired());
