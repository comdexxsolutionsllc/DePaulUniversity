00005:    This class carries out the merge sort algorithm.
00006: */
00007: public class MergeSorter

00009:    /**
00010:       Sorts an array, using the merge sort algorithm.
00011:       @param a the array to sort
00012:       @param comp the comparator to compare array elements
00013:    */
00014:    public static <E> void sort(E[] a, Comparator<? super E> comp)
00015:    {
00016:       mergeSort(a, 0, a.length - 1, comp);
00017:    }
00018: 
00019:    /**
00020:       Sorts a range of an array, using the merge sort
00021:       algorithm.
00022:       @param a the array to sort
00023:       @param from the first index of the range to sort
00024:       @param to the last index of the range to sort
00025:       @param comp the comparator to compare array elements
00026:    */
00027:    private static <E> void mergeSort(E[] a, int from, int to,
00028:       Comparator<? super E> comp)
00029:    {
00030:       if (from == to) return;
00031:       int mid = (from + to) / 2;
00032:       // Sort the first and the second half
00033:       mergeSort(a, from, mid, comp);
00034:       mergeSort(a, mid + 1, to, comp);
00035:       merge(a, from, mid, to, comp);
00036:    }
00037: 
00038:    /**
00039:       Merges two adjacent subranges of an array
00040:       @param a the array with entries to be merged
00041:       @param from the index of the first element of the
00042:          first range
00043:       @param mid the index of the last element of the
00044:          first range
00045:       @param to the index of the last element of the
00046:          second range
00047:       @param comp the comparator to compare array elements
00048:    */
00049:    private static <E> void merge(E[] a,
00050:       int from, int mid, int to, Comparator<? super E> comp)
00051:    {
00052:       int n = to - from + 1;
00053:          // Size of the range to be merged
00054: 
00055:       // Merge both halves into a temporary array b
00056:       ArrayList<E> b = new ArrayList<E>(n);
00057: 
00058:       int i1 = from;
00059:          // Next element to consider in the first range
00060:       int i2 = mid + 1;
00061:          // Next element to consider in the second range
00062:       int j = 0;
00063:          // Next open position in b
00064: 
00065:       // As long as neither i1 nor i2 past the end, move
00066:       // the smaller element into b
00067:       while (i1 <= mid && i2 <= to)
00068:       {
00069:          if (comp.compare(a[i1], a[i2]) < 0)
00070:          {
00071:             b.set(j,a[i1]);
00072:             i1++;
00073:          }
00074:          else
00075:          {
00076:             b.set(j,a[i2]);
00077:             i2++;
00078:          }
00079:          j++;
00080:       }
00081: 
00082:       // Note that only one of the two while loops
00083:       // below is executed
00084: 
00085:       // Copy any remaining entries of the first half
00086:       while (i1 <= mid)
00087:       {
00088:          b.set(j,a[i1]);
00089:          i1++;
00090:          j++;
00091:       }
00092: 
00093:       // Copy any remaining entries of the second half
00094:       while (i2 <= to)
00095:       {
00096:          b.set(j,a[i2]);
00097:          i2++;
00098:          j++;
00099:       }
00100: 
00101:       // Copy back from the temporary array
00102:       for (j = 0; j < n; j++)
00103:         a[from + j] = b.get(j);
00104:    }
