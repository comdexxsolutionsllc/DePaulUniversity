00183:       if (center)
00184:       {
00185:          if (p.getX() > q.getX()) 
00186:          { 
00187:             return getAttachmentPoint(g2, q, p, arrow, d, center); 
00188:          }
00189:          attach = new Point2D.Double((p.getX() + q.getX()) / 2, 
00190:             (p.getY() + q.getY()) / 2);
00191:          if (p.getY() < q.getY())
00192:             yoff =  - GAP - d.getHeight();
00193:          else if (p.getY() == q.getY())
00194:             xoff = -d.getWidth() / 2;
00195:          else
00196:             yoff = GAP;
00197:       }
00198:       else 
00199:       {
00200:          if (p.getX() < q.getX())
00201:          {
00202:             xoff = -GAP - d.getWidth();
00203:          }
00204:          if (p.getY() > q.getY())
00205:          {
00206:             yoff = GAP;
00207:          }
00208:          if (arrow != null)
00209:          {
00210:             Rectangle2D arrowBounds = arrow.getPath(p, q).getBounds2D();
00211:             if (p.getX() < q.getX())
00212:             {
00213:                xoff -= arrowBounds.getWidth();
00214:             }
00215:             else
00216:             {
00217:                xoff += arrowBounds.getWidth();
00218:             }
00219:          }
00220:       }
00221:       return new Point2D.Double(attach.getX() + xoff, attach.getY() + yoff);
