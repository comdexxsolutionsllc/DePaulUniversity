00010:    A class that assumes that an edge can yield its shape

00017:       Returns the path that should be stroked to

00025:    {
00026:       return getShape().getBounds();
00027:    }
00028: 
00029:    public boolean contains(Point2D aPoint)
00030:    {
00031:       final double MAX_DIST = 3;
00032: 
00033:       // the end points may contain small nodes, so don't
00034:       // match them
00035:       Line2D conn = getConnectionPoints();
00036:       if (aPoint.distance(conn.getP1()) <= MAX_DIST 
00037:          || aPoint.distance(conn.getP2()) <= MAX_DIST)
00038:          return false;
00039: 
00040:       Shape p = getShape();
00041:       BasicStroke fatStroke = new BasicStroke(
00042:          (float)(2 * MAX_DIST));
00043:       Shape fatPath = fatStroke.createStrokedShape(p);
00044:       return fatPath.contains(aPoint);
00045:    }
