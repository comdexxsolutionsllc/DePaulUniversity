104      /**
105       * Implementation of Record interface.
106       *
107       * <p>This is a utility class for Inventory.  Fields are mutable and
108       * package-private.</p>
109       *
110       * <p><b>Class Invariant:</b> No two instances may reference the same Video.</p>
111       *
112       * @see Record
113       */
114      private static final class RecordObj implements Record {
115        Video video; // the video
116        int numOwned;   // copies owned
117        int numOut;     // copies currently rented
118        int numRentals; // total times video has been rented
119        
120        RecordObj(Video video, int numOwned, int numOut, int numRentals) {
121          this.video = video;
122          this.numOwned = numOwned;
123          this.numOut = numOut;
124          this.numRentals = numRentals;
125        }
126        public Video video() {
127          return video;
128        }
129        public int numOwned() {
130          return numOwned;
131        }
132        public int numOut() {
133          return numOut;
134        }
135        public int numRentals() {
136          return numRentals;
137        }
138        public String toString() {
139          StringBuilder buffer = new StringBuilder();
140          buffer.append(video);
141          buffer.append(" [");
142          buffer.append(numOwned);
143          buffer.append(",");
144          buffer.append(numOut);
145          buffer.append(",");
146          buffer.append(numRentals);
147          buffer.append("]");
148          return buffer.toString();
149        }
150      }
